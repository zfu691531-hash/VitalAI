// 输入预处理管道（Pipeline）
class InputPreprocessor {

  async process(raw_input, input_type, user_id) {
    // Step1: 语音转文字（如果是语音输入）
    let text = raw_input
    if (input_type === "voice") {
      text = await asr.transcribe(raw_input)  // ASR语音识别
      // 王奶奶常见口音纠错：把"血压"识别成"学压"等
      text = await applyUserAccentCorrection(text, user_id)
    }

    // Step2: 文本规范化
    text = text
      .replace(/[！？。，、；：]/g, match => punctMap[match])  // 全角→半角
      .replace(/s+/g, " ")         // 多余空格
      .trim()

    // Step3: 语境融合（加入对话历史，帮助理解代词和省略）
    const context = await hotDB.get(`context:${user_id}`)  // 热数据层读取对话上下文
    const enriched = {
      text,
      context_window: context?.recent_turns || [],  // 最近3轮对话
      user_state:     context?.user_state,           // 当前用户状态
      timestamp:      new Date().toISOString(),
    }

    // Step4: 安全过滤（隐私卫士Agent鉴权）
    const safe = await privacyGuard.check(enriched)
    if (!safe.ok) return { blocked: true, reason: safe.reason }

    return enriched
  }
}

【原始输入（语音）】"那个...血压的事，我还要继续吃那个药不？"

【规范化】"那个 血压的事 我还要继续吃那个药不" → 去掉停顿词后：
          "血压的事 我还要继续吃那个药不"

# Bert微调训练脚本（Python/PyTorch）

from transformers import BertForSequenceClassification, BertTokenizer, Trainer, TrainingArguments
import torch

# 使用中文Bert预训练模型
MODEL_NAME = "bert-base-chinese"
NUM_LABELS = 15  # 意图类别数量（7大类×子类）

class IntentDataset(torch.utils.data.Dataset):
    def __init__(self, texts, labels, tokenizer, max_length=128):
        self.encodings = tokenizer(
            texts,
            truncation=True,
            padding=True,
            max_length=max_length,
            return_tensors="pt"
        )
        self.labels = torch.tensor(labels)

    def __getitem__(self, idx):
        return {k: v[idx] for k, v in self.encodings.items()}, self.labels[idx]

    def __len__(self): return len(self.labels)

# 训练配置
training_args = TrainingArguments(
    output_dir       = "./intent_model",
    num_train_epochs = 5,          # 5轮，防止过拟合
    per_device_batch = 32,
    learning_rate    = 2e-5,       # Bert微调标准学习率
    warmup_steps     = 100,
    weight_decay     = 0.01,
    save_strategy    = "epoch",
    evaluation_strategy = "epoch",
    load_best_model  = True,
)

# 模型初始化
model = BertForSequenceClassification.from_pretrained(
    MODEL_NAME,
    num_labels=NUM_LABELS
)

# 训练
trainer = Trainer(
    model     = model,
    args      = training_args,
    train_dataset = train_dataset,
    eval_dataset  = eval_dataset,
)
trainer.train()

// Bert推理服务（Node.js调用Python推理API）
class BertIntentClassifier {

  async classify(text, context_window) {
    // 构造输入：文本 + 最近2轮对话拼接（增强语境理解）
    const input = context_window.length > 0
      ? `[CTX] ${context_window.slice(-2).join(" [SEP] ")} [NOW] ${text}`
      : text

    const response = await fetch("http://bert-service:8000/classify", {
      method:  "POST",
      headers: {"Content-Type": "application/json"},
      body:    JSON.stringify({ text: input }),
      signal:  AbortSignal.timeout(200)  // 200ms超时，超时直接降级到大模型
    })

    const result = await response.json()
    // result = {
    //   intent:        "HEALTH_QUERY.bp_query",
    //   confidence:     0.93,
    //   all_scores:    { "HEALTH_QUERY.bp_query": 0.93, "SIMPLE_QUERY": 0.05, ... }
    //   is_multi:       false,  // 是否检测到多意图信号
    // }

    return result
  }
}

// LLM意图拆分服务
class LLMIntentDecomposer {

  async decompose(preprocessed_input, user_context) {
    const { text, context_window, user_state } = preprocessed_input

    // 构造动态System Prompt（注入用户实时状态）
    const system_prompt = INTENT_DECOMPOSE_SYSTEM_PROMPT
      .replace("{user_state}",    JSON.stringify(user_state))
      .replace("{current_time}",  new Date().toLocaleString("zh-CN"))
      .replace("{context_window}", context_window.slice(-3).join(" | "))

    // 调用大模型
    const response = await fetch("https://api.anthropic.com/v1/messages", {
      method:  "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model:      "claude-sonnet-4-6",
        max_tokens: 1000,
        system:     system_prompt,
        messages: [{ role: "user", content: text }]
      })
    })

    const data  = await response.json()
    const raw   = data.content[0].text

    // 解析JSON输出
    let result
    try {
      result = JSON.parse(raw)
    } catch (e) {
      // JSON解析失败：启用兜底解析（正则提取关键字段）
      result = this.fallbackParse(raw)
    }

    // 安全校验：确保所有必要字段存在
    return this.validate(result)
  }

  // 兜底解析：LLM输出格式不标准时的处理
  fallbackParse(raw_text) {
    console.warn("LLM输出格式异常，启用兜底解析")
    // 从文本中提取关键信息，构造最低可用的意图结构
    return {
      intent_count: 1,
      intents: [{ type: "UNKNOWN", agent: "agent_supervisor",
                   description: raw_text.slice(0, 100) }],
      execution_order: ["intent_1"],
      needs_confirmation: true,
      confirmation_message: "奶奶，我不太确定您的意思，您能再说一遍吗？"
    }
  }
}

// 意图路由层：IntentRouter
class IntentRouter {

  async route(decomposed_result, user_id, trace_id) {
    const { intents, execution_order, needs_confirmation } = decomposed_result

    // Step1: 安全前置检查
    const safety_pass = await this.safetyCheck(intents)
    if (!safety_pass.ok) {
      return this.handleSafetyBlock(safety_pass, user_id)
    }

    // Step2: 如果需要用户确认（LLM判断的）
    if (needs_confirmation) {
      await notifyAgent.speak(decomposed_result.confirmation_message)
      const confirmed = await waitUserConfirmation(user_id, timeout=30000)
      if (!confirmed) return { status: "cancelled_by_user" }
    }

    // Step3: 按执行顺序提交到冲突仲裁层（先提交意图声明）
    const intent_declarations = intents.map(intent => ({
      intent_id:        intent.id,
      agent_id:         AGENT_MAP[intent.agent],
      action:           intent.type,
      goal_weight:      GOAL_WEIGHTS[intent.type],
      resources_needed: getRequiredResources(intent),
      flexibility:      { time_shift: "±15min", can_merge: true },
    }))
    await conflictDetector.submit(intent_declarations)  // 冲突检测

    // Step4: 按顺序执行（处理依赖关系）
    const results = {}
    for (const intent_id of execution_order) {
      const intent = intents.find(i => i.id === intent_id)

      // 等待依赖意图完成
      if (intent.dependencies?.length > 0) {
        await waitForDependencies(intent.dependencies, results)
      }

      // 向对应Agent下发标准指令（通过消息队列）
      const cmd = this.buildCommand(intent, results, trace_id)
      await msgQueue.dispatch({
        to:       AGENT_MAP[intent.agent],
        priority: intent.priority,
        msg_type: "COMMAND",
        trace_id: trace_id,  // 全链路追踪
        payload:  cmd
      })

      // 等待Agent执行完成（L1反馈事件）
      results[intent_id] = await waitForAgentResult(intent.id, timeout=10000)
    }

    // Step5: 综合所有结果，生成最终回复
    return this.synthesizeResponse(results, decomposed_result)
  }

  // Agent映射表：意图类型 → 实际Agent ID
  AGENT_MAP = {
    "健康模块Agent": "agent_health",
    "日常管理Agent": "agent_daily",
    "出行Agent":     "agent_trip",
    "精神模块Agent": "agent_mental",
    "Tool Agent":    "agent_tool",
  }

  // 目标权重：与冲突仲裁层对齐
  GOAL_WEIGHTS = {
    "HEALTH_QUERY":   0.85,
    "TRIP_REQUEST":   0.70,
    "MENTAL_CONNECT": 0.75,
    "DAILY_TASK":     0.70,
    "SIMPLE_QUERY":   0.50,
  }
}

// 安全前置检查：防止意图被滥用
async function safetyCheck(intents) {
  for (const intent of intents) {

    // 检查1：外出意图 + 健康风险同时存在时，必须有健康检查依赖
    if (intent.type === "TRIP_REQUEST") {
      const has_health_check = intents.some(i =>
        i.type === "HEALTH_QUERY" &&
        intent.dependencies?.includes(i.id)  // 外出必须依赖健康检查
      )
      if (!has_health_check) {
        // 自动补充健康检查意图
        intents.unshift({
          id: "auto_health_check",
          type: "HEALTH_QUERY",
          agent: "健康模块Agent",
          priority: 1,
          description: "外出前自动健康检查",
        })
        intent.dependencies = [...(intent.dependencies||[]), "auto_health_check"]
      }
    }

    // 检查2：用药相关指令，必须验证用户身份
    if (intent.type === "HEALTH_QUERY" && intent.subtype?.includes("medication_change")) {
      return { ok: false, reason: "用药调整需要家属或医生确认，系统无法执行" }
    }

    // 检查3：隐私相关数据，隐私卫士Agent鉴权
    if (intent.key_params?.includes_sensitive_data) {
      const auth = await privacyGuard.authorize(intent, user_id)
      if (!auth.ok) return { ok: false, reason: auth.reason }
    }
  }

  return { ok: true }
}

// 完整意图处理主入口
async function handleUserInput(raw_input, input_type, user_id) {
  const trace_id = generateTraceId()  // 全链路追踪ID
  const start_time = Date.now()

  try {
    // ── 阶段1：预处理（~20ms）──
    const preprocessed = await preprocessor.process(raw_input, input_type, user_id)
    if (preprocessed.blocked) return respondBlocked(preprocessed.reason)

    // ── 阶段2：第一层 Bert快速识别（~50ms）──
    const bert_result = await bertClassifier.classify(
      preprocessed.text,
      preprocessed.context_window
    )

    let final_intents

    // ── 阶段3：路由决策 ──
    const needs_llm = bert_result.confidence < 0.85 || bert_result.is_multi

    if (!needs_llm) {
      // 高置信单意图 → 直接构造结构化意图，跳过大模型
      final_intents = buildSimpleIntent(bert_result)
      console.log(`[FAST PATH] ${Date.now()-start_time}ms`)

    } else {
      // 低置信或多意图 → 升级第二层大模型（~1000~2000ms）
      final_intents = await llmDecomposer.decompose(preprocessed, { user_id })
      console.log(`[LLM PATH] ${Date.now()-start_time}ms`)
    }

    // ── 阶段4：路由执行（依赖通信层+冲突层）──
    const exec_result = await intentRouter.route(final_intents, user_id, trace_id)

    // ── 阶段5：反馈回路（触发L1即时反馈）──
    await feedbackCollector.record({
      trace_id,
      input:        preprocessed.text,
      bert_result,
      final_intents,
      exec_result,
      total_ms:     Date.now() - start_time,
      bert_correct: null,  // 后续用户行为验证后填入
    })

    return exec_result

  } catch (error) {
    // ── 全局异常兜底 ──
    return handleException(error, user_id, trace_id)
  }
}

// 持续学习数据收集器
class ContinualLearningCollector {

  // 每次意图处理后，收集用于改进模型的数据
  async collect(trace_id, input_text, bert_prediction, actual_outcome) {

    const training_sample = {
      text:           input_text,
      bert_predicted: bert_prediction.intent,
      bert_confidence:bert_prediction.confidence,
      // 实际结果来源：
      // 1. 用户对执行结果的反馈（USER_APPROVE/REJECT）
      // 2. LLM的拆分结果（当Bert不自信时，LLM的结果是真值）
      // 3. 中置信度执行后的追问确认
      actual_intent:  actual_outcome.confirmed_intent,
      is_correct:     bert_prediction.intent === actual_outcome.confirmed_intent,
      recorded_at:    new Date().toISOString(),
    }

    // 存入温数据层，等待定期微调
    await warmDB.insert("training_samples", training_sample)

    // 如果Bert严重错误（预测和真实差距大），记录为高优先级样本
    if (!training_sample.is_correct && bert_prediction.confidence > 0.8) {
      // 高置信但错误：最需要修正的样本
      await warmDB.flagHighPriority("training_samples", training_sample)
      console.warn(`高置信误判：输入="${input_text}" 预测=${bert_prediction.intent} 实际=${actual_outcome.confirmed_intent}`)
    }
  }

  // 每周触发一次增量微调
  async weeklyFinetune() {
    const new_samples = await warmDB.query({
      table: "training_samples",
      recorded_at_gte: lastWeek(),
      min_count: 50  // 至少50条新样本才触发微调
    })

    if (new_samples.length >= 50) {
      await bertTrainer.incrementalFinetune(new_samples)
      console.log(`周度微调完成：新增${new_samples.length}条样本`)
    }
  }
}
