// 所有数据出境（发给外部API/工具）前必须经过隐私卫士处理
class PrivacyGuard {

  async sanitizeForExternalCall(data, destination) {
    const sanitized = { ...data }

    // Level1：直接标识符脱敏
    sanitized.user_name    = this.pseudonymize(data.user_name)   // "王奶奶"→"用户001"
    sanitized.phone        = this.mask(data.phone, 4)            // "138XXXX"→"****5678"
    sanitized.address      = this.fuzzify(data.address)          // 精确地址→区级模糊
    sanitized.birthday     = this.ageRange(data.birthday)        // 1953年→"70-75岁"

    // Level2：准标识符泛化（防止重识别攻击）
    if (sanitized.bp_value && destination !== "hospital_api") {
      sanitized.bp_value = this.roundTo5(sanitized.bp_value)  // 152→150（精度降低）
    }

    // Level3：敏感属性特殊处理
    // 过敏史、慢性病：永远不发给非医疗机构
    if (!this.isMedicalPartner(destination)) {
      delete sanitized.allergies
      delete sanitized.chronic_conditions
    }

    // 记录审计日志
    await auditLog.record({
      action: "external_call",
      destination,
      data_types: Object.keys(data),
      sanitized_fields: this.getDiff(data, sanitized)
    })

    return sanitized
  }

  // AI输出内容审查：防止AI给出不安全建议
  async auditAIOutput(output, context) {
    const risks = []

    // 检查1：是否包含具体用药建议（不能说"您应该增加降压药剂量"）
    if (/增加.*剂量|停止服用|换.*药/.test(output)) {
      risks.push({ type: "medical_advice", severity: "high" })
      output = output.replace(/增加.*剂量|停止服用/g, "建议咨询您的医生")
    }

    // 检查2：是否包含确定性的预测（不能说"您可能患有xxx病"）
    if (/您患有|这是.*病/.test(output)) {
      risks.push({ type: "medical_diagnosis", severity: "critical" })
    }

    // 检查3：是否承诺了无法兑现的事（不能说"我会永远陪着您"）
    if (/永远|一定会|保证/.test(output)) {
      risks.push({ type: "false_promise", severity: "medium" })
    }

    return { safe: risks.length === 0, risks, sanitized_output: output }
  }
}

// 差分隐私：为统计查询添加Laplace噪声
function addDifferentialPrivacy(true_value, sensitivity, epsilon=1.0) {
  // epsilon越小→隐私保护越强（但噪声越大）
  // 对老年健康数据，使用epsilon=0.5，较强保护
  const noise_scale = sensitivity / epsilon
  const noise = laplaceSample(0, noise_scale)
  return true_value + noise
}
