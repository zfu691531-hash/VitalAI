一、现状诊断
图谱中个人信息库节点连接了：日常管理Agent、健康模块Agent、精神模块总Agent、智能汇报Agent、隐私卫士Agent、数据流转模块。每一个都可以直接读写，没有任何路由规则和隔离机制。

1.1 三类读写混在一层的真实问题


1.2 RAG 和 Knowledge Graph 的边界不清晰
图谱中 RAG 和 Neo4j图数据库 同时存在，但没有说明：什么数据放 RAG？什么放图数据库？查询时走哪个？当两者数据矛盾时信哪个？


二、三层分级存储架构
参考大型互联网系统的"冷热数据分离"设计，把个人信息库拆成三层。每层有独立的存储介质、独立的读写路径、独立的生命周期管理。



2.1 热数据层详细设计
热数据层是所有Agent日常操作的"工作台"。它必须快、准、新鲜。


2.2 温数据层详细设计
温数据层是"近期记忆"，存放过去3个月的行为模式、偏好数据和L2聚合结果。它不需要毫秒级响应，但要支持"时间范围查询"和"趋势分析"。

2.3 冷数据层详细设计：RAG 与 Knowledge Graph 边界
冷数据层是"长期记忆"，包含两个独立的子系统，职责完全不同，绝不混用。




三、统一查询路由层
所有Agent不再直接访问具体的存储层，而是统一通过"查询路由层（Data Router）"访问。路由层根据查询类型、时间范围、数据类型自动决定走哪一层。




四、并发写入保护与数据生命周期
4.1 乐观锁防止多Agent并发写入冲突
图谱中精神模块Agent、智能汇报Agent、反思Agent都可能同时更新同一个字段（如情绪状态）。乐观锁通过版本号机制确保不丢数据。

4.2 数据生命周期管道：热→温→冷自动迁移

// 健康Agent读取示例（目标：<50ms）
async function getLatestVitals(userId) {
  const cached = await redis.get(`hot:${userId}:vitals`)
  if (cached && !isExpired(cached)) return cached  // 命中缓存
  // 未命中：从PostgreSQL读，写回Redis
  const fresh = await pg.query("SELECT * FROM vitals WHERE user_id=$1 ORDER BY measured_at DESC LIMIT 1", [userId])
  await redis.setex(`hot:${userId}:vitals`, 3600, fresh)
  return fresh
}

// Data Router：统一查询入口
class DataRouter {

  async query(request) {
    const { user_id, data_type, time_range, query_text, need_graph } = request

    // ── 路由规则1：需要图关系 → Neo4j ──
    if (need_graph || GRAPH_TYPES.includes(data_type)) {
      return this.queryKnowledgeGraph(request)
    }

    // ── 路由规则2：语义搜索 → RAG ──
    if (query_text && !time_range) {
      return this.queryRAG(request)
    }

    // ── 路由规则3：按时间范围路由 ──
    const daysAgo = getDaysAgo(time_range?.start)
    if (!daysAgo || daysAgo <= 7) {
      return this.queryHot(request)       // 近7天 → 热数据层
    } else if (daysAgo <= 90) {
      return this.queryWarm(request)      // 近3个月 → 温数据层
    } else {
      return this.queryCold(request)      // 更早 → 冷数据层
    }
  }

  // 冲突解决：热数据和冷数据结论矛盾时
  async resolveConflict(hot_result, cold_result) {
    // 原则：热数据（近7天行为）优先于冷数据（历史记录）
    // 例外：医疗禁忌（过敏史）以冷数据（图数据库）为准，不可被近期行为覆盖
    if (cold_result.is_medical_constraint) return cold_result
    return hot_result  // 近期行为更能反映当前实际状态
  }
}

// 乐观锁写入流程（完整实现）
class HotDataWriter {

  async write(user_id, data_type, new_value, agent_id) {
    // Step1：读取当前版本
    const current = await hotDB.get(`${user_id}:${data_type}`)
    // current = { value: {...}, version: 42 }

    // Step2：业务处理（这里可能需要几秒）
    const processed = await this.process(new_value, current.value)

    // Step3：CAS写入（Compare And Set）
    const result = await hotDB.compareAndSet({
      key:              `${user_id}:${data_type}`,
      expected_version: current.version,   // 我读到的版本
      new_value:        processed,
      new_version:      current.version + 1,
      written_by:       agent_id,
      written_at:       new Date().toISOString()
    })

    if (result.success) {
      console.log(`版本${current.version}→${current.version+1}，写入成功`)
      return result
    }

    if (result.error === "VERSION_CONFLICT") {
      // 版本冲突：读取对方写入的内容，执行合并策略
      const their_value = result.current_value
      const merged = await this.merge(processed, their_value, data_type)
      // 递归重试（带退避，最多3次）
      return this.writeWithRetry(user_id, data_type, merged, agent_id, 1)
    }
  }

  merge(my_val, their_val, data_type) {
    switch(data_type) {
      case "emotion_status":
        // 情绪状态：取两份分析的均值，保留两个来源的原始评分
        return {
          score: (my_val.score + their_val.score) / 2,
          sources: [...(their_val.sources || []), { agent: my_val.agent, score: my_val.score }]
        }
      case "medication_status":
        // 用药状态：以最新的确认时间为准（不能丢失"已服药"状态）
        return my_val.confirmed_at > their_val.confirmed_at ? my_val : their_val
      default:
        return my_val  // 默认：后写优先
    }
  }
}

// 数据迁移管道（定时运行）
class DataMigrationPipeline {

  // 每天凌晨2点运行：热数据→温数据迁移
  async migrateHotToWarm() {
    const expired = await hotDB.scan({ ttl_lt: 0 })  // 找到所有已过期的热数据
    for (const item of expired) {
      // 过期热数据归档到温数据层（PostgreSQL）
      await warmDB.insert({
        user_id:     item.user_id,
        data_type:   item.data_type,
        value:       item.value,
        recorded_at: item.updated_at,
        expires_at:  addDays(item.updated_at, 90)  // 温数据保留3个月
      })
      await hotDB.delete(item.key)
    }
    console.log(`热→温迁移完成：${expired.length}条`)
  }

  // 每月1日运行：温数据→冷数据迁移
  async migrateWarmToCold() {
    const old_records = await warmDB.query({
      expires_at_lt: new Date()  // 超过3个月的温数据
    })
    for (const record of old_records) {
      // 判断是否有长期参考价值
      if (hasLongTermValue(record)) {
        // 有价值→送入RAG向量化+归档
        await ragDB.upsert({ text: summarize(record), metadata: record })
      }
      // 无论如何都从温数据层删除，释放空间
      await warmDB.delete(record.id)
    }
  }
}
