// MAB接入：Thompson Sampling算法（适合小样本·每个用户独立学习）

// ── 定义"臂"：一天8个推送时间段 ──
const TIME_SLOTS = [
  "07:00-08:30",  // 晨起时段
  "08:30-10:00",  // 早饭后
  "10:00-11:30",  // 上午活跃
  "11:30-13:00",  // 午饭前后
  "13:00-15:00",  // 午休时段
  "15:00-16:30",  // 下午茶时段
  "16:30-18:00",  // 傍晚活跃
  "18:00-20:00",  // 晚饭时段
]

// ── 每个用户·每类提醒·独立维护Beta分布参数 ──
const mab_state = {
  user_id: "wang001",
  reminder_type: "water",  // 喝水提醒
  arms: [
    // {alpha: 成功次数+1, beta: 失败次数+1} -- Beta分布参数
    { slot:"07:00-08:30", alpha:1,  beta:8  },  // 响应率低
    { slot:"10:00-11:30", alpha:5,  beta:4  },  // 中等
    { slot:"15:00-16:30", alpha:19, beta:6  },  // 最高！
    // ... 其他时段
  ]
}

// ── Thompson Sampling：选择推送时段 ──
function selectBestSlot(arms) {
  // 每个臂从Beta分布中采样一个值
  const samples = arms.map(arm => ({
    slot:   arm.slot,
    sample: betaSample(arm.alpha, arm.beta)  // 随机采样
  }))
  // 选择采样值最高的臂（有概率探索非最优臂）
  return samples.sort((a,b) => b.sample - a.sample)[0].slot
}

// ── 推送完成后更新参数 ──
function updateArm(arms, chosen_slot, responded) {
  const arm = arms.find(a => a.slot === chosen_slot)
  if (responded) arm.alpha += 1  // 成功：alpha+1
  else           arm.beta  += 1  // 失败：beta+1
  // 自动持久化到温数据层
  warmDB.upsert({ user_id, reminder_type, arms })
}

// ── 接入推送调度器 ──
class PushScheduler {
  async schedule(user_id, reminder_type, reminder_content) {
    const state = await warmDB.get(`mab:${user_id}:${reminder_type}`)
    const best_slot = selectBestSlot(state.arms)

    // 在最优时段安排推送
    await scheduler.enqueue({
      content:  reminder_content,
      send_at:  getNextSlotStart(best_slot),
      callback: (responded) => updateArm(state.arms, best_slot, responded)
    })
    console.log(`MAB选择时段：${best_slot}，当前预估成功率：${state.arms.find(a=>a.slot===best_slot).alpha/(state.arms.find(a=>a.slot===best_slot).alpha+state.arms.find(a=>a.slot===best_slot).beta)*100}%`)
  }
}

// MAB降级：数据不足时或模型异常时的兜底
function selectWithFallback(arms, reminder_type) {
  const total_pulls = arms.reduce((sum, a) => sum + a.alpha + a.beta - 2, 0)

  if (total_pulls < 20) {
    // 数据不足20次：使用均匀随机探索（cold start）
    return TIME_SLOTS[Math.floor(Math.random() * TIME_SLOTS.length)]
  }

  try {
    return selectBestSlot(arms)
  } catch (e) {
    // MAB计算异常：回退到经验默认时段
    const SAFE_DEFAULTS = {
      water:      "15:00-16:30",
      medication: "08:30-10:00",
      exercise:   "10:00-11:30",
    }
    return SAFE_DEFAULTS[reminder_type] || "10:00-11:30"
  }
}

// Fuzzy Logic：血压风险模糊评估

// ── 步骤1：定义模糊集合（隶属度函数）──
// 收缩压（SBP）的五个模糊集合
function sbpMembership(sbp) {
  return {
    // 低血压：<90时隶属度1，90-110线性降至0
    low:     sbp < 90  ? 1 : sbp < 110 ? (110-sbp)/20 : 0,
    // 正常：100-130完全正常，两侧平滑过渡
    normal:  sbp < 100 ? 0 : sbp < 120 ? (sbp-100)/20 :
             sbp < 130 ? 1 : sbp < 145 ? (145-sbp)/15 : 0,
    // 正常偏高（高风险预警区）：130-150
    elevated:sbp < 130 ? 0 : sbp < 140 ? (sbp-130)/10 :
             sbp < 150 ? (150-sbp)/10 : 0,
    // 高血压1期：140-160
    high1:   sbp < 140 ? 0 : sbp < 160 ? (sbp-140)/20 :
             sbp < 170 ? (170-sbp)/10 : 0,
    // 高血压危象：>160
    crisis:  sbp < 160 ? 0 : sbp < 180 ? (sbp-160)/20 : 1,
  }
}

// ── 步骤2：定义规则库 ──
const FUZZY_RULES = [
  // { if SBP is X AND DBP is Y, then risk is Z }
  { sbp:"normal",   dbp:"normal",   risk:"low",      weight:1.0 },
  { sbp:"elevated", dbp:"normal",   risk:"moderate", weight:0.8 },
  { sbp:"normal",   dbp:"elevated", risk:"moderate", weight:0.9 },  // 舒张压高更重要
  { sbp:"high1",    dbp:"any",      risk:"high",     weight:1.0 },
  { sbp:"crisis",   dbp:"any",      risk:"critical", weight:1.0 },
  { sbp:"elevated", dbp:"elevated", risk:"high",     weight:0.95 }, // 双高协同
]

// ── 步骤3：个体基线校正 ──
function adjustForBaseline(sbp, user_baseline) {
  // 王奶奶平时血压偏低，基线120/75
  // 当前138对她来说比对普通人的138更危险
  const deviation = sbp - user_baseline.sbp_avg
  return sbp + deviation * 0.3  // 偏差校正系数0.3
}

// ── 步骤4：综合评估，输出风险评分 0-1 ──
async function evaluateBloodPressure(sbp, dbp, user_id) {
  // 从热数据层获取个人基线
  const baseline = await hotDB.get(`${user_id}:bp_baseline`)
  const adj_sbp = adjustForBaseline(sbp, baseline)

  const sbp_mem = sbpMembership(adj_sbp)
  const dbp_mem = dbpMembership(dbp)

  // 应用规则，计算各风险等级的激活度
  const risk_scores = { low:0, moderate:0, high:0, critical:0 }
  for (const rule of FUZZY_RULES) {
    const activation = Math.min(
      sbp_mem[rule.sbp] || 1,
      dbp_mem[rule.dbp] || 1
    ) * rule.weight
    risk_scores[rule.risk] = Math.max(risk_scores[rule.risk], activation)
  }

  // 去模糊化（重心法）→ 输出0-1风险评分
  const risk_score = defuzzify(risk_scores)

  return {
    risk_score,                           // 0-1连续值，而非0/1跳变
    risk_level: getRiskLevel(risk_score), // low/moderate/high/critical
    intervention: getIntervention(risk_score),
    explanation: `收缩压${adj_sbp}(含基线校正)·风险评分${risk_score.toFixed(2)}`
  }
}

// ── 步骤5：根据风险评分决定中断级别 ──
// 替换原来的 if(sbp > 145) trigger_P1()
function triggerByRisk(risk_result) {
  if (risk_result.risk_score > 0.85) return "P0"         // 高血压危象
  if (risk_result.risk_score > 0.65) return "P1"         // 明确高风险
  if (risk_result.risk_score > 0.45) return "WATCH"      // 持续观察，不触发中断
  return "NORMAL"                                         // 正常
}

// HMM步态异常检测：完整接入设计

// ── 状态空间定义 ──
const GAIT_STATES = ["normal", "cautious", "unsteady", "fall_risk"]

// ── 观测空间：从传感器读取的特征 ──
// 每步提取：步长、步频、左右对称性、加速度方差、落脚冲击力
function extractGaitFeatures(raw_sensor_data) {
  return {
    stride_length:   calcStrideLength(raw_sensor_data),
    cadence:         calcCadence(raw_sensor_data),
    symmetry_index:  calcSymmetry(raw_sensor_data),   // 0=完全对称，1=严重不对称
    acc_variance:    calcAccVariance(raw_sensor_data), // 加速度方差
    impact_force:    calcImpactForce(raw_sensor_data),
  }
}

// ── 触发时机：三种情况触发HMM分析 ──
class GaitMonitor {
  constructor() {
    this.window_size = 20   // 连续20步作为一个分析窗口
    this.step_buffer = []   // 步态特征缓冲区
    this.hmm_model   = loadPretrainedHMM()  // 预训练的HMM模型（离线训练）
  }

  async onNewStep(step_data) {
    const features = extractGaitFeatures(step_data)
    this.step_buffer.push(features)

    // 触发1：滑动窗口满20步，定期分析
    if (this.step_buffer.length >= this.window_size) {
      await this.analyze()
      this.step_buffer = this.step_buffer.slice(-10)  // 保留后10步滚动
    }

    // 触发2：单步加速度方差突增（快速响应，不等窗口满）
    if (features.acc_variance > SUDDEN_CHANGE_THRESHOLD) {
      await this.analyzeUrgent()
    }
  }

  async analyze() {
    // Viterbi算法：找最可能的隐藏状态序列
    const state_sequence = viterbi(this.hmm_model, this.step_buffer)
    const current_state  = state_sequence[state_sequence.length - 1]
    const trend          = detectTrend(state_sequence)  // 是否在恶化

    if (current_state === "fall_risk" || trend === "deteriorating") {
      // 结果流向：推送到Event Aggregator → Decision Core
      await eventBus.publish("health.gait.anomaly", {
        user_id:        "wang001",
        hmm_state:      current_state,
        confidence:     calcConfidence(state_sequence),  // 0-1
        state_sequence: state_sequence,  // 完整状态序列，供智能汇报Agent分析
        trend:          trend,
        timestamp:      new Date().toISOString()
      })
    }
  }
}

// PSO出行规划：完整接入设计

// ── 问题定义 ──
// 场景：王奶奶明天要：① 上午去社区医院复查（9:00-11:00）
//               ② 顺路去菜市场买菜（不超过1小时）
//               ③ 下午去老年大学上课（14:00-16:00）
//               ④ 回家（偏好：距离近、不爬楼梯、不转多次车）

// 粒子：一种行程安排（各地点的访问顺序+交通方式）
// 目标函数：最小化（总距离×0.3 + 等待时间×0.3 + 体力消耗×0.4）
// 体力消耗权重最高（老人优先考虑体力）

class PSOTripPlanner {

  plan(destinations, constraints, user_preference) {
    // 初始化粒子群（每个粒子=一种行程方案）
    const particles = Array(50).fill(null).map(() => ({
      position: randomShuffle(destinations),  // 随机访问顺序
      velocity: randomVelocity(),
      best_pos: null,
      best_fit: Infinity
    }))

    let global_best = { position: null, fitness: Infinity }

    // 迭代优化（最多100次，或适应度<0.1时提前终止）
    for (let iter = 0; iter < 100; iter++) {
      for (const p of particles) {
        // 计算适应度（考虑时间窗口约束、体力消耗、距离）
        const fitness = calcFitness(p.position, constraints, user_preference)

        if (fitness < p.best_fit) {
          p.best_pos = [...p.position]
          p.best_fit = fitness
        }
        if (fitness < global_best.fitness) {
          global_best = { position: [...p.position], fitness }
        }
      }

      // 更新粒子速度和位置（PSO核心公式）
      updateParticles(particles, global_best)

      if (global_best.fitness < 0.1) break  // 早停
    }

    return {
      route:    global_best.position,
      fitness:  global_best.fitness,
      eta:      calculateETA(global_best.position),
      transport:chooseTransport(global_best.position, user_preference),
    }
  }

  // 失败降级：PSO超时或无解时的兜底
  fallback(destinations) {
    // 简单贪心：按时间窗口顺序排列
    return destinations.sort((a,b) => a.time_window.start - b.time_window.start)
  }
}

// 体力消耗评估（老人专属）
function calcPhysicalCost(route, user_data) {
  return route.reduce((cost, dest, i) => {
    const segment_distance = calcDistance(route[i-1], dest)
    const stairs = dest.has_stairs ? 2.0 : 1.0    // 有台阶×2
    const transfer = dest.bus_transfers * 0.5       // 每次换乘+0.5
    return cost + segment_distance * stairs + transfer
  }, 0)
}

    }

  task_backlog:   ["empty", "normal", "heavy"],       // 积压任务量
  user_state:     ["awake", "resting", "sleeping"],
}
// 状态数 = 3×3×3×3 = 81种状态

// ── 动作空间 ──
// 每一步决策：下一个要执行的任务类型
const ACTIONS = ["health_monitor", "medication_remind", "exercise_guide",
                 "emotional_support", "daily_task", "social_connect", "wait"]

// ── 奖励函数 ──
function getReward(action, state, outcome) {
  let reward = 0
  // 正向奖励
  if (outcome.user_responded)     reward += 2.0   // 用户响应了
  if (outcome.health_improved)    reward += 3.0   // 健康指标改善
  if (outcome.user_satisfied > 0.7) reward += 1.5 // 用户满意度高
  // 负向惩罚
  if (outcome.user_ignored)       reward -= 1.0   // 用户忽略了
  if (outcome.p1_triggered)       reward -= 5.0   // 未预防的健康危机
  if (outcome.user_complained)    reward -= 3.0   // 用户明确抱怨
  // 长期价值
  if (action === "health_monitor" && state.health_risk === "high") reward += 1.0
  return reward
}

// ── Q-Table更新（DiDierVI简化版）──
function updateQTable(state, action, reward, next_state) {
  const current_q = Q_TABLE[state][action]
  const max_next_q = Math.max(...Object.values(Q_TABLE[next_state]))
  // 贝尔曼方程
  Q_TABLE[state][action] = current_q +
    LEARNING_RATE * (reward + DISCOUNT_FACTOR * max_next_q - current_q)
}

// ── 接入Supervisor决策流程 ──
// 当Decision Core需要决定"下一步执行什么任务"时调用
function selectNextAction(system_state) {
  const state_key = encodeState(system_state)

  // ε-贪婪：90%选最优，10%随机探索（保持学习能力）
  if (Math.random() < EPSILON) {
    return ACTIONS[Math.floor(Math.random() * ACTIONS.length)]
  }

  const q_values = Q_TABLE[state_key]
  return Object.entries(q_values).sort((a,b)=>b[1]-a[1])[0][0]
}
